ğŸ§© Module in Action

import {AudioPro} from 'react-native-audio-pro';

AudioPro.configure({ contentType: 'speech' });

// Works with static remote files, live streams, and local files
const track = {
  id: 'track-001',
  url: 'https://example.com/audio.mp3', // Remote file: 'https://example.com/livestream' or local file
  title: 'Sample Track',
  artwork: 'https://example.com/image.jpg', // Remote image or local image
  artist: 'Narrator Name',
};

// Play with default options
AudioPro.play(track);

// With custom options
AudioPro.play(track, {
  autoPlay: false,
  headers: {
    audio: { 'Authorization': 'Bearer token123' },
    artwork: { 'Cache-Control': 'max-age=3600' }
  }
});
